test_output <- function(old_func, new_func) {
    expected <- old_func()
    actual <- new_func()
    test <- tryCatch(all(expected == actual),
                     error=function(cond) FALSE)
    if (test) {
        message("Success!")
    } else {
        message("Failure...")
        message("Expected:")
        print(expected)
        message("Actual:") 
        print(actual)
    }
}
